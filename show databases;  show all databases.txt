CREATE TABLE civisol (
    `id` INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    `nom` varchar(500) NOT NULL,
    `email` varchar(500) NOT NULL,
    `adresse` varchar(500) NOT NULL,
    `postal` varchar(500) NOT NULL,
    `ville` varchar(500) NOT NULL,
    `solaires` varchar(500) NOT NULL,
    `couverture_value` varchar(500) NOT NULL,
    `souhaitee` varchar(500) NOT NULL,
    `profondeur` varchar(500) NOT NULL,
    `envisage` varchar(500) NOT NULL,
    `orientation_value` varchar(500) NOT NULL,
    `panneaux` varchar(500) NOT NULL,
    `central` varchar(500) NOT NULL,
    `distance` varchar(500) NOT NULL,
    `courant_value` varchar(500) NOT NULL,
    `complementaires` varchar(500) NOT NULL,
    `photos` varchar(500) NOT NULL
);
